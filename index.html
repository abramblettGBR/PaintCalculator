  <script>

    let roomCount = 1;


    function showExportButtons() {

      document.getElementById('export-button').classList.remove('hidden');

      document.getElementById('copy-button').classList.remove('hidden');

    }


    document.getElementById('add-room').addEventListener('click', () => {

      roomCount++;

      const roomsContainer = document.getElementById('rooms');

      const templateRoom = document.querySelector('.room');

      const newRoom = templateRoom.cloneNode(true);

      newRoom.dataset.roomId = roomCount;

      newRoom.querySelector('h2').textContent = `Room ${roomCount}`;

      // Reset form fields

      newRoom.querySelectorAll('input, select').forEach(el => {

        if (el.type === 'checkbox') el.checked = true;

        else if (el.type === 'number') el.value = el.defaultValue || '';

        else if (el.tagName === 'SELECT') el.selectedIndex = 0;

        else el.value = '';

      });

      roomsContainer.appendChild(newRoom);

      updateRemoveButtons();

    });


    document.getElementById('rooms').addEventListener('click', (e) => {

      if (e.target.classList.contains('remove-room')) {

        e.target.closest('.room').remove();

        updateRoomNumbers();

        updateRemoveButtons();

      }

    });


    function updateRoomNumbers() {

      document.querySelectorAll('.room').forEach((room, index) => {

        room.dataset.roomId = index + 1;

        room.querySelector('h2').textContent = `Room ${index + 1}`;

      });

      roomCount = document.querySelectorAll('.room').length;

    }


    function updateRemoveButtons() {

      document.querySelectorAll('.remove-room').forEach(btn => {

        btn.classList.toggle('hidden', roomCount <= 1);

      });

    }


    document.getElementById('calculate').addEventListener('click', () => {

      showExportButtons();


      const rooms = document.querySelectorAll('.room');

      let totalPrimerGallons = 0;

      let totalFinishGallons = 0;

      let paintTotals = {}; // finish paint by color/sheen


      const doorOpeningDeduct = 20;   // sq ft deducted from walls per door

      const doorFaceArea = 21;        // sq ft per side for painting

      const primerCoverage = 250;

      const finishCoverage = 350;


      let resultHTML = '<p class="text-xl font-bold text-gray-800 mb-4">Paint & Primer Estimate</p>';

      let exportContent = 'GBR Paint Calculator Results\n\n';


      rooms.forEach((room, index) => {

        const label = room.querySelector('.room-label').value || `Room ${index + 1}`;

        const length = parseFloat(room.querySelector('.length').value) || 0;

        const width = parseFloat(room.querySelector('.width').value) || 0;

        const height = parseFloat(room.querySelector('.height').value) || 0;

        const numDoors = parseInt(room.querySelector('.doors').value) || 0;

        const numWindows = parseInt(room.querySelector('.windows').value) || 0;

        const trimLength = parseFloat(room.querySelector('.trim').value) || 0;


        const includePrimer = room.querySelector('.include-primer').checked;

        const primerCoats = parseInt(room.querySelector('.primer-coats').value) || 1;

        const finishCoats = parseInt(room.querySelector('.finish-coats').value) || 2;

        const wasteFactor = (parseFloat(room.querySelector('.waste-factor').value) || 10) / 100;


        const wallColor = room.querySelector('.wall-color').value || 'Unnamed';

        const wallSheen = room.querySelector('.wall-sheen').value || 'None';

        const ceilingColor = room.querySelector('.ceiling-color').value || 'Unnamed';

        const ceilingSheen = room.querySelector('.ceiling-sheen').value || 'None';

        const trimColor = room.querySelector('.trim-color').value || 'Unnamed';

        const trimSheen = room.querySelector('.trim-sheen').value || 'None';


        const paintDoors = room.querySelector('.paint-doors').value;


        // Areas

        const wallGross = 2 * (length * height + width * height);

        const deductions = (numDoors * doorOpeningDeduct) + (numWindows * 15);

        const wallNet = Math.max(0, wallGross - deductions);

        const ceilingArea = length * width;


        // Primer (walls + ceiling)

        let primerGallons = 0;

        if (includePrimer && primerCoats > 0) {

          const primerArea = (wallNet + ceilingArea) * primerCoats;

          const rawPrimer = primerArea / primerCoverage;

          primerGallons = Math.ceil(rawPrimer * (1 + wasteFactor));

          totalPrimerGallons += primerGallons;

        }


        // Finish paint

        const wallFinishRaw = (wallNet * finishCoats) / finishCoverage;

        const ceilingFinishRaw = (ceilingArea * finishCoats) / finishCoverage;

        let trimFinishRaw = (trimLength * finishCoats) / finishCoverage;


        // Doors (same paint as trim)

        let doorsFinishRaw = 0;

        let doorsPaintedSqFt = 0;

        if (paintDoors !== 'none' && numDoors > 0) {

          const sides = paintDoors === 'both' ? 2 : 1;

          doorsPaintedSqFt = numDoors * doorFaceArea * sides;

          doorsFinishRaw = (doorsPaintedSqFt * finishCoats) / finishCoverage;

          trimFinishRaw += doorsFinishRaw;

        }


        const finishGallonsRaw = wallFinishRaw + ceilingFinishRaw + trimFinishRaw;

        const finishGallons = Math.ceil(finishGallonsRaw * (1 + wasteFactor));

        totalFinishGallons += finishGallons;


        // Group finish paint

        if (wallFinishRaw > 0) {

          const key = `${wallColor} Walls (${wallSheen})`;

          paintTotals[key] = (paintTotals[key] || 0) + wallFinishRaw;

        }

        if (ceilingFinishRaw > 0) {

          const key = `${ceilingColor} Ceiling (${ceilingSheen})`;

          paintTotals[key] = (paintTotals[key] || 0) + ceilingFinishRaw;

        }

        if (trimFinishRaw > 0) {

          const suffix = doorsFinishRaw > 0 ? ' (incl. Doors)' : '';

          const key = `${trimColor} Trim${suffix} (${trimSheen})`;

          paintTotals[key] = (paintTotals[key] || 0) + trimFinishRaw;

        }


        // Room output

        resultHTML += `

          <div class="mb-6 border-b pb-4 last:border-b-0">

            <p class="text-lg font-semibold text-gray-800">${label}:</p>

            <p class="text-gray-700">Walls net: ${wallNet.toFixed(1)} sq ft</p>

            <p class="text-gray-700">Ceiling: ${ceilingArea.toFixed(1)} sq ft</p>

            ${includePrimer ? `<p class="text-lime-700">Primer: ${primerGallons} gallons (${primerCoats} coat${primerCoats !== 1 ? 's' : ''})</p>` : '<p class="text-gray-500">Primer: Skipped</p>'}

            <p class="text-indigo-700 font-medium">Finish paint: $$   {finishGallons} gallons (   $${finishCoats} coat${finishCoats !== 1 ? 's' : ''})</p>

            <p class="text-sm text-gray-600 mt-1">

              • Walls: ${wallFinishRaw.toFixed(2)} gal<br>

              • Ceiling: ${ceilingFinishRaw.toFixed(2)} gal<br>

              • Trim${doorsFinishRaw > 0 ? ' + Doors' : ''}: ${trimFinishRaw.toFixed(2)} gal (pre-waste)

            </p>

          </div>

        `;


        exportContent += `${label}:\n`;

        exportContent += `  Walls net area: ${wallNet.toFixed(1)} sq ft\n`;

        exportContent += `  Ceiling area: ${ceilingArea.toFixed(1)} sq ft\n`;

        if (includePrimer) exportContent += `  Primer: ${primerGallons} gallons (${primerCoats} coats)\n`;

        exportContent += `  Finish paint: ${finishGallons} gallons (${finishCoats} coats)\n`;

        exportContent += `    - Walls: ${wallFinishRaw.toFixed(2)} gal\n`;

        exportContent += `    - Ceiling: ${ceilingFinishRaw.toFixed(2)} gal\n`;

        exportContent += `    - Trim${doorsFinishRaw > 0 ? ' + Doors' : ''}: ${trimFinishRaw.toFixed(2)} gal\n\n`;

      });


      const grandTotal = totalPrimerGallons + totalFinishGallons;


      resultHTML += `

        <div class="mt-8 p-6 bg-indigo-100 rounded-xl">

          <p class="text-2xl font-bold text-indigo-800">Project Totals</p>

          <p class="text-xl mt-2">Primer: <span class="text-lime-700">${totalPrimerGallons} gallons</span></p>

          <p class="text-xl">Finish Paint: <span class="text-indigo-700">${totalFinishGallons} gallons</span></p>

          <p class="text-3xl font-bold mt-4">Grand Total: ${grandTotal} gallons</p>

        </div>


        <p class="mt-6 text-lg font-semibold text-gray-800">Finish Paint Summary by Color/Sheen:</p>

      `;


      exportContent += `PROJECT TOTALS:\n`;

      exportContent += `  Primer: ${totalPrimerGallons} gallons\n`;

      exportContent += `  Finish Paint: ${totalFinishGallons} gallons\n`;

      exportContent += `  Grand Total: ${grandTotal} gallons\n\n`;

      exportContent += `Finish Paint by Color/Sheen (pre-waste):\n`;


      for (const [key, amount] of Object.entries(paintTotals)) {

        const gal = amount.toFixed(2);

        resultHTML += `<p class="text-gray-700">${key}: ${gal} gallons</p>`;

        exportContent += `  ${key}: ${gal} gallons\n`;

      }


      resultHTML += `

        <p class="text-sm text-gray-500 mt-8">

          Assumptions: Primer 250 sq ft/gal, Finish 350 sq ft/gal, Door opening deduct 20 sq ft, Door face 21 sq ft/side, Window deduct 15 sq ft.<br>

          Always verify with your paint product data sheets.

        </p>

      `;


      exportContent += `\nAssumptions:\n  Primer ~250 sq ft/gal\n  Finish ~350 sq ft/gal\n  Door opening deduct: 20 sq ft\n  Door face area: 21 sq ft per side\n  Window deduct: 15 sq ft\n`;


      document.getElementById('result').innerHTML = resultHTML;


      // Export / Copy handlers

      document.getElementById('export-button').onclick = () => {

        const blob = new Blob([exportContent], { type: 'text/plain' });

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url;

        a.download = 'gbr_paint_calculator_results.txt';

        a.click();

        URL.revokeObjectURL(url);

      };


      document.getElementById('copy-button').onclick = () => {

        navigator.clipboard.writeText(exportContent).then(() => {

          alert('Results copied to clipboard!');

        }).catch(() => {

          alert('Copy failed. Try the download button.');

        });

      };

    });


    updateRemoveButtons();

  </script>
